Performance of short transmission line, determination of voltage regulation and efficiency

EXPERIMENT 5

FORMATION OF Y-BUS FOR POWER SYSTEM BY INSPECTION METHOD

clc

clear

// F T Z HLC

linedata = [1 2 0.2*(%i) 0.03*(%i);

1 4 0.25*(%i) 0.025*(%i);

2 3 0.4*(%i) 0.02*(%i);

2 4 0.5*(%i) 0.015*(%i);

3 4 0.2*(%i) 0.01*(%i)];

Sh=[4 0.02*(%i)]; // b-,

from_bus=real(linedata(:,1)); //

to_bus=real(linedata(:,2)); //

Z=linedata(:,3); //

HLC=linedata(:,4);

nbus=max(max(from_bus),max(to_bus));

e=length(from_bus);

Ybus=zeros(nbus, nbus);

for k=1:e // k takes values from 1 to maximum line number e

p=from_bus(k);

q=to_bus(k);

z=Z(k);

hlc=HLC(k);

y=1/z; // y = reciprocal of z

Ybus(p,q)=Ybus(p,q)-y;

Ybus(q,p)=Ybus(p,q);

// formation of the diagonal elements

Ybus(p,p)=Ybus(p,p)+y+hlc;

Ybus(q,q)=Ybus(q,q)+y+hlc;

end

// Addition of shunt admittance data

// Bus no. at which shunt compensation is provided

p=Sh(1,1);

sh_compensation=Sh(1,2);

p=real(p);

Ybus(p,p)=Ybus(p,p)+sh_compensation ;

Ybus

EXPERIMENT 6

FORMATION OF Y-BUS FOR POWER SYSTEMWITHOUT MUTUAL COUPLING, BY SINGULAR TRANSFORMATION METHOD

clc

clear

// F T Z HLC

linedata = [1 2 0.2*(%i) 0.03*(%i);

1 4 0.25*(%i) 0.025*(%i);

2 3 0.4*(%i) 0.02*(%i);

2 4 0.5*(%i) 0.015*(%i);

3 4 0.2*(%i) 0.01*(%i)];

from_bus=real(linedata(:,1));

to_bus=real(linedata(:,2));

Z=linedata(:,3);

HLC=linedata(:,4);

nbus=max(max(from_bus),max(to_bus));

e=length(from_bus);

a=zeros(e,nbus);

Zpri=zeros(e,e);

for k=1:e // k takes values from 1 to maximum line number e

p=from_bus(k,1);

q=to_bus(k,1);

z=linedata(k,3);

//Formation of Bus Incidence matrix

a(k,p)=1;

a(k,q)=-1;

Zpri(k,k)=z;

end

Ypri=inv(Zpri);

Ybus=a'*Ypri*a;

//Addition of Half Line Charging Admittance to diagonal elements of Ybus

for k=1:1:e

p=from_bus(k,1);

q=to_bus(k,1);

hlc=HLC(k,1);

Ybus(p,p)=Ybus(p,p)+hlc;

Ybus(q,q)=Ybus(q,q)+hlc;

end

disp(Ybus)

ECONOMIC DISPATCH IN POWER SYSTEM NEGLECTING LOSSES

Pmin = constraint (: ,1);

Pmax = constraint (: ,2);

chk = zeros (n ,1) ;

rem =1;

while rem ==1

sx =0; sy =0;

for i =1: n

if i~= chk (i)

sx=sx+b(i) /(2* a(i));

sy=sy +1/(2* a(i));

end

end

lamda =( pd+sx)/sy;

sch =0;

for i =1: n

if i~= chk (i)

P(i)=( lamda -b(i)) /(2* a(i));

if P(i)Pmax (i)

if P(i)< Pmin (i)

P(i)= Pmin (i);

else

P(i)= Pmax (i);

end

pd=pd -P(i);

chk (i)=i;

sch = sch +1;

end

end

if sch ==0

rem =0;

else

rem =1;

end

end
